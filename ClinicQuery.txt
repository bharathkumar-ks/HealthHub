CREATE TABLE CLINIC_TB (
    RECORDID VARCHAR2(12 BYTE) PRIMARY KEY,
    PATIENTNAME VARCHAR2(40 BYTE) NOT NULL,
    DOCTORNAME VARCHAR2(40 BYTE) NOT NULL,
    APPOINTMENT_DATE DATE NOT NULL,
    CONSULTATION_TYPE VARCHAR2(30 BYTE),
    FEES NUMBER,
    REMARKS VARCHAR2(100 BYTE)
);


CREATE SEQUENCE CLINIC_SEQ
START WITH 10
INCREMENT BY 1
MAXVALUE 99
NOCYCLE;
